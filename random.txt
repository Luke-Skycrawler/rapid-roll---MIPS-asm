    addi r1,zero,16807
    mult heap,r1
    mflo r1
    mfhi r2
    nor r3,r1,zero
    add heap,r1,r2
    slt r3,r2,r3
    bne r3,zero,label_random_1
        addi heap,heap,1
    label_random_1:
jr $ra
